<script>
    gsap.registerPlugin(MotionPathPlugin);

    const COAST_DATA = [
        { id:'kl1',loc:'基隆嶼海域',x:280,y:40,p:0.45, story:'這裡有你小時候看過的海龜，因為你的自備，今天減少了塑膠進入。謝謝你！', trivia:'龍蝦棲息地。', img:'https://images.unsplash.com/photo-1505118380757-91f5f45d8de4?w=800', shop:'小島咖啡', type:'Friendly Cafe', simg:'https://images.unsplash.com/photo-1509042239860-f550ce710b93?w=100' },
        { id:'td1',loc:'台東三仙台',x:260,y:460,p:0.38, story:'太平洋的第一道曙光，因為你的守護而更加明亮。', trivia:'八拱橋地標。', img:'https://images.unsplash.com/photo-1520113526514-99dec0cb393d?w=800', shop:'成功民宿', type:'Green Inn', simg:'https://images.unsplash.com/photo-1566073771259-6a8506099945?w=100' },
        { id:'ph1',loc:'澎湖七美',x:40,y:300,p:0.52, story:'雙心石滬的智慧，需要我們這一代繼續傳承守護。', trivia:'石滬傳統漁法。', img:'https://images.unsplash.com/photo-1538332576228-eb5b4c4de6f5?w=800', shop:'七美舍', type:'Local Hub', simg:'https://images.unsplash.com/photo-1555396273-367ea4eb4db5?w=100' }
    ];

    let unlocked = JSON.parse(localStorage.getItem('muse_unlocked_v27')) || [];
    let stamps = parseInt(localStorage.getItem('muse_stamps_v27')) || 0;
    let current = null;

    window.enterApp = () => {
        gsap.to("#mirror-gate", { opacity: 0, scale: 2, duration: 1.5, ease: "power4.inOut", onComplete: () => {
            document.getElementById('mirror-gate').style.display = 'none';
            gsap.to("#app-shell", { opacity: 1, duration: 1 });
            updateStampUI();
            if (localStorage.getItem('drawn_v27')) lockDraw();
        }});
    };

    // --- 敘事節奏動畫 ---
    window.startAction = () => {
        if (localStorage.getItem('drawn_v27')) return;
        current = COAST_DATA[Math.floor(Math.random()*COAST_DATA.length)];
        const overlay = document.getElementById('anim-overlay');
        overlay.style.display = 'flex';
        
        const photon = document.getElementById('light-essence');
        const tl = gsap.timeline();

        // 1. 光芒出發
        gsap.to("#line-1", { opacity: 1, y: -10, duration: 1 });
        
        tl.fromTo(photon, { top: "90%", left: "50%", scale: 1, opacity: 1 }, { 
            motionPath: { path: "#river-path", align: "#river-path", alignOrigin: [0.5, 0.5] },
            duration: 3, ease: "power2.inOut" 
        })
        .add(() => {
            gsap.to("#line-1", { opacity: 0, duration: 0.5 });
            gsap.to("#line-2", { opacity: 1, y: -10, delay: 0.5, duration: 1 });
        })
        .to(photon, { scale: 50, opacity: 0, duration: 0.8, delay: 0.5 })
        .add(() => {
            gsap.to("#line-2", { opacity: 0, duration: 0.5 });
            const line3 = document.getElementById('line-3');
            line3.innerText = "你現在守護的是：" + current.loc;
            gsap.to(line3, { opacity: 1, y: -10, duration: 1 });
            document.getElementById('btn-reveal-final').style.display = 'inline-block';
        });
    };

    window.revealReward = () => {
        gsap.to("#anim-overlay", { opacity: 0, duration: 0.8, onComplete: () => {
            document.getElementById('anim-overlay').style.display = 'none';
            document.getElementById('anim-overlay').style.opacity = 1;
            document.querySelectorAll('.narrative-line').forEach(l => l.style.opacity = 0);
            document.getElementById('btn-reveal-final').style.display = 'none';
            showCard();
        }});
    };

    function showCard() {
        document.getElementById('empty-hint').style.display = 'none';
        document.getElementById('card-3d').style.display = 'block';
        document.getElementById('c-img').src = current.img;
        document.getElementById('c-loc').innerText = current.loc;
        document.getElementById('c-story').innerText = current.story;
        document.getElementById('c-impact').innerText = `此次行動減少了 ${current.p}g 塑膠進入這片海。${current.trivia}`;
        document.getElementById('c-s-name').innerText = current.shop;
        document.getElementById('c-s-type').innerText = current.type;
        document.getElementById('c-s-img').src = current.simg;

        gsap.fromTo("#card-3d", { scale: 0.7, opacity: 0 }, { scale: 1, opacity: 1, duration: 1.2, ease: "elastic.out(1, 0.6)" });
        localStorage.setItem('drawn_v27', 'true');
        lockDraw();
    }

    function lockDraw() {
        const btn = document.getElementById('draw-btn');
        btn.classList.add('disabled');
        btn.innerText = "今日守護已達成";
    }

    // --- 印章與入庫 ---
    window.saveToArchive = (e) => {
        e.stopPropagation();
        const card = document.getElementById('card-3d');
        const icon = document.getElementById('archive-icon');
        const rect = icon.getBoundingClientRect();

        stamps++; if(stamps > 5) stamps = 5;
        localStorage.setItem('muse_stamps_v27', stamps);
        updateStampUI();

        gsap.to(card, {
            x: rect.left - card.offsetLeft - 150,
            y: rect.top - card.offsetTop - 250,
            scale: 0, opacity: 0, duration: 1.2, ease: "power3.inOut",
            onComplete: () => {
                card.style.display = 'none';
                icon.classList.add('highlight');
                setTimeout(() => icon.classList.remove('highlight'), 500);
                document.getElementById('empty-hint').innerHTML = `<i class="fa-solid fa-circle-check" style="font-size:4rem; color:var(--accent); margin-bottom:20px;"></i><p>感謝守護，碎片已入庫</p>`;
                if(!unlocked.includes(current.id)) unlocked.push(current.id);
                localStorage.setItem('muse_unlocked_v27', JSON.stringify(unlocked));
            }
        });
    };

    window.toggleSidebar = (on) => {
        const shell = document.getElementById('app-shell');
        const mask = document.getElementById('sidebar-mask');
        if(on) { mask.style.display='block'; gsap.to(mask,{opacity:1}); shell.classList.add('shell-spatial'); document.body.classList.add('sidebar-open'); }
        else { gsap.to(mask,{opacity:0, onComplete:()=>mask.style.display='none'}); shell.classList.remove('shell-spatial'); document.body.classList.remove('sidebar-open'); }
    };

    function updateStampUI() {
        const slots = document.querySelectorAll('.stamp-slot');
        slots.forEach((s, i) => { if(i < stamps) s.classList.add('active'); });
    }

    window.toggleGallery = (on) => {
        const m = document.getElementById('map-modal');
        if(on) { m.style.display='flex'; renderMap(); gsap.from(m,{y:"100%"}); }
        else { gsap.to(m,{y:"100%", onComplete:()=> m.style.display='none'}); }
    };

    function renderMap() {
        const svg = document.getElementById('taiwan-map');
        svg.querySelectorAll('.m-dot').forEach(d => d.remove());
        COAST_DATA.forEach(p => {
            const has = unlocked.includes(p.id);
            const c = document.createElementNS("http://www.w3.org/2000/svg", "circle");
            c.setAttribute("cx", p.x); c.setAttribute("cy", p.y);
            c.setAttribute("r", has ? 10 : 5);
            c.setAttribute("class", `m-dot`);
            c.style.fill = has ? 'var(--accent)' : '#111';
            c.style.stroke = has ? '#fff' : 'rgba(255,255,255,0.1)';
            svg.appendChild(c);
        });
    }

    window.resetData = () => {
        if(confirm("重置所有數據？")) { localStorage.clear(); location.reload(); }
    };

    window.saveAvatar = (i) => {
        const reader = new FileReader();
        reader.onload = (e) => document.getElementById('u-avatar').src = e.target.result;
        reader.readAsDataURL(i.files[0]);
    };

    window.flipCard = () => { const inner = document.getElementById('card-inner'); const rot = gsap.getProperty(inner, "rotationY"); gsap.to(inner, { rotationY: rot === 180 ? 0 : 180, duration: 0.6 }); };
</script>
